<script>
(function(){
  var f=document.getElementById('icon-frame');
  var ro=null;
  function fit(){ try{ f.style.height=f.contentWindow.document.documentElement.scrollHeight+'px'; }catch(e){} }
  function watch(){
    fit();
    try{
      var doc=f.contentWindow.document;
      if(ro) ro.disconnect();
      ro=new ResizeObserver(fit);
      ro.observe(doc.documentElement);
      ro.observe(doc.body);
    }catch(e){}
  }
  window.addEventListener('resize', fit);
  f.addEventListener('load', watch);
  if(f.contentWindow && f.contentWindow.document.readyState==='complete') watch();
})();
</script>
