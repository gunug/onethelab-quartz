<iframe id="icon-frame" src="/static/icon/icon_builder.html" scrolling="no" style="width:100%; border:0; display:block;" loading="lazy"></iframe>

<script>
(function(){
  var f=document.getElementById('icon-frame');
  if(!f) return;
  function onMsg(e){
    if(e.source!==f.contentWindow) return;
    var d=e.data;
    if(d && d.type==='iconFrameHeight' && d.height){ f.style.height=d.height+'px'; }
  }
  window.addEventListener('message', onMsg);
})();
</script>
